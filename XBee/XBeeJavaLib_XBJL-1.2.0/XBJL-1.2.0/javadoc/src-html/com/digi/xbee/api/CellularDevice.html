547        /**
548         * Sends the provided SMS message to the given phone number.
549         * 
550         * <p>This method blocks till a success or error response arrives or the 
551         * configured receive timeout expires.</p>
552         * 
553         * <p>The receive timeout is configured using the {@code setReceiveTimeout}
554         * method and can be consulted with {@code getReceiveTimeout} method.</p>
555         * 
556         * <p>For non-blocking operations use the method 
557         * {@link #sendSMSAsync(String, String)}.</p>
558         * 
559         * @param phoneNumber The phone number to send the SMS to.
560         * @param data String containing the text of the SMS.
561         * 
562         * @throws IllegalArgumentException if {@code phoneNumber.length() > 20} or
563         *                                  if {@code phoneNumber} is invalid.
564         * @throws InterfaceNotOpenException if this device connection is not open.
565         * @throws NullPointerException if {@code phoneNumber == null} or 
566         *                              if {@code data == null}.
567         * @throws TimeoutException if there is a timeout sending the SMS.
568         * @throws XBeeException if there is any other XBee related exception.
569         * 
570         * @see #getReceiveTimeout()
571         * @see #setReceiveTimeout(int)
572         * @see #sendSMSAsync(String, String)
573         */
574        public void sendSMS(String phoneNumber, String data) throws TimeoutException, XBeeException {
575                if (phoneNumber == null)
576                        throw new NullPointerException("Phone number cannot be null");
577                if (data == null)
578                        throw new NullPointerException("Data cannot be null");
579                
580                // Check if device is remote.
581                if (isRemote())
582                        throw new OperationNotSupportedException("Cannot send SMS from a remote device.");
583                
584                logger.debug(toString() + "Sending SMS to {} >> {}.", phoneNumber, data);
585                
586                XBeePacket xbeePacket = new TXSMSPacket(getNextFrameID(), phoneNumber, data);
587                
588                sendAndCheckXBeePacket(xbeePacket, false);
589        }
590        
591        /**
592         * Sends asynchronously the provided SMS to the given phone number.
593         * 
594         * <p>Asynchronous transmissions do not wait for answer or for transmit 
595         * status packet.</p>
596         * 
597         * @param phoneNumber The phone number to send the SMS to.
598         * @param data String containing the text of the SMS.
599         * 
600         * @throws IllegalArgumentException if {@code phoneNumber.length() > 20} or
601         *                                  if {@code phoneNumber} is invalid.
602         * @throws InterfaceNotOpenException if this device connection is not open.
603         * @throws NullPointerException if {@code phoneNumber == null} or 
604         *                              if {@code data == null}.
605         * @throws TimeoutException if there is a timeout sending the SMS.
606         * @throws XBeeException if there is any other XBee related exception.
607         * 
608         * @see #sendSMS(String, String)
609         */
610        public void sendSMSAsync(String phoneNumber, String data) throws TimeoutException, XBeeException {
611                if (phoneNumber == null)
612                        throw new NullPointerException("Phone number cannot be null");
613                if (data == null)
614                        throw new NullPointerException("Data cannot be null");
615                
616                // Check if device is remote.
617                if (isRemote())
618                        throw new OperationNotSupportedException("Cannot send SMS from a remote device.");
619                
620                logger.debug(toString() + "Sending SMS asynchronously to {} >> {}.", phoneNumber, data);
621                
622                XBeePacket xbeePacket = new TXSMSPacket(getNextFrameID(), phoneNumber, data);
623                
624                sendAndCheckXBeePacket(xbeePacket, true);
625        }
