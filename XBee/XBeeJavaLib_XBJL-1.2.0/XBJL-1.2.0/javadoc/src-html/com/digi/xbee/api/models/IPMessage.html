016package com.digi.xbee.api.models;
017
018import java.net.Inet4Address;
019
020/**
021 * This class represents an IP message containing the IP address the 
022 * message belongs to, the source and destination ports, the IP protocol,
023 * and the content (data) of the message. 
024 * 
025 * <p>This class is used within the XBee Java Library to read data sent to IP 
026 * devices.</p>
027 * 
028 * @since 1.2.0
029 */
030public class IPMessage {
031
032        // Variables.
033        private final Inet4Address ipAddress;
034        
035        private final byte[] data;
036        
037        private final int sourcePort;
038        private final int destPort;
039        
040        private final IPProtocol protocol;
041        
042        /**
043         * Class constructor. Instantiates a new object of type 
044         * {@code IPMessage} with the given parameters.
045         * 
046         * @param ipAddress The IP address the message comes from.
047         * @param sourcePort TCP or UDP source port of the transmission.
048         * @param destPort TCP or UDP destination port of the transmission.
049         * @param protocol IP protocol used in the transmission.
050         * @param data Byte array containing the data of the message.
051         * 
052         * @throws IllegalArgumentException if {@code sourcePort < 0} or
053         *                                  if {@code sourcePort > 65535} or
054         *                                  if {@code destPort < 0} or
055         *                                  if {@code destPort > 65535}.
056         * @throws NullPointerException if {@code ipAddress == null} or
057         *                              if {@code data == null} or
058         *                              if {@code protocol ==  null}.
059         * 
060         * @see com.digi.xbee.api.models.IPProtocol
061         * @see java.net.Inet4Address
062         */
063        public IPMessage(Inet4Address ipAddress, int sourcePort, int destPort, 
064                        IPProtocol protocol, byte[] data) {
065                if (ipAddress == null)
066                        throw new NullPointerException("IP address cannot be null.");
067                if (protocol == null)
068                        throw new NullPointerException("Protocol cannot be null.");
069                if (data == null)
070                        throw new NullPointerException("Data cannot be null.");
071                
072                if (sourcePort < 0 || sourcePort > 65535)
073                        throw new IllegalArgumentException("Source port must be between 0 and 65535.");
074                if (destPort < 0 || destPort > 65535)
075                        throw new IllegalArgumentException("Destination port must be between 0 and 65535.");
076                
077                this.ipAddress = ipAddress;
078                this.sourcePort = sourcePort;
079                this.destPort = destPort;
080                this.protocol = protocol;
081                this.data = data;
082        }
083        
084        /**
085         * Returns the IP address this message is associated to.
086         * 
087         * @return The IP address this message is associated to.
088         * 
089         * @see java.net.Inet4Address
090         */
091        public Inet4Address getIPAddress() {
092                return ipAddress;
093        }
094        
095        /**
096         * Returns the source port of the transmission.
097         * 
098         * @return The source port of the transmission.
099         */
100        public int getSourcePort() {
101                return sourcePort;
102        }
103        
104        /**
105         * Returns the destination port of the transmission.
106         * 
107         * @return The destination port of the transmission.
108         */
109        public int getDestPort() {
110                return destPort;
111        }
112        
113        /**
114         * Returns the protocol used in the transmission.
115         * 
116         * @return The protocol used in the transmission
117         * 
118         * @see IPProtocol
119         */
120        public IPProtocol getProtocol() {
121                return protocol;
122        }
123        
124        /**
125         * Returns the byte array containing the data of the message.
126         * 
127         * @return A byte array containing the data of the message.
128         */
129        public byte[] getData() {
130                return data;
131        }
132        
133        /**
134         * Returns the data of the message in string format.
135         * 
136         * @return The data of the message in string format.
137         */
138        public String getDataString() {
139                return new String(data);
140        }
141}
