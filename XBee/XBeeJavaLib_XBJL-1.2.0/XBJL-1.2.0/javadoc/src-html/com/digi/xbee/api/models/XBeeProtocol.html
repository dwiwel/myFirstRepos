016package com.digi.xbee.api.models;
017
018import java.util.HashMap;
019
020/**
021 * Enumerates the available XBee protocols. The XBee protocol is determined 
022 * by the combination of hardware and firmware of an XBee device.
023 */
024public enum XBeeProtocol {
025        
026        // Enumeration entries
027        ZIGBEE(0, "ZigBee"),
028        RAW_802_15_4(1, "802.15.4"),
029        XBEE_WIFI(2, "Wi-Fi"),
030        DIGI_MESH(3, "DigiMesh"),
031        XCITE(4, "XCite"),
032        XTEND(5, "XTend (Legacy)"),
033        XTEND_DM(6, "XTend (DigiMesh)"),
034        SMART_ENERGY(7, "Smart Energy"),
035        DIGI_POINT(8, "Point-to-multipoint"),
036        ZNET(9, "ZNet 2.5"),
037        XC(10, "XSC"),
038        XLR(11, "XLR"),
039        XLR_DM(12, "XLR"), // TODO [XLR_DM] XLR device with DigiMesh support.
040        SX(13, "XBee SX"),
041        XLR_MODULE(14, "XLR Module"),
042        /** @since 1.2.0 */
043        CELLULAR(15, "Cellular"),
044        UNKNOWN(99, "Unknown");
045        
046        // Variables
047        private static final HashMap<Integer, XBeeProtocol> lookupTable = new HashMap<Integer, XBeeProtocol>();
048        
049        private final int id;
050        
051        private final String description;
052        
053        static {
054                for (XBeeProtocol xbeeProtocol:values())
055                        lookupTable.put(xbeeProtocol.getID(), xbeeProtocol);
056        }
057        
058        /**
059         * Class constructor. Instantiates a new {@code XBeeProtocol} enumeration 
060         * entry with the given parameters.
061         * 
062         * @param id XBee protocol ID.
063         * @param description XBee protocol description.
064         */
065        private XBeeProtocol(int id, String description) {
066                this.id = id;
067                this.description = description;
068        }
069        
070        /**
071         * Returns the XBee protocol ID.
072         * 
073         * @return XBee protocol ID.
074         */
075        public int getID() {
076                return id;
077        }
078        
079        /**
080         * Returns the XBee protocol description.
081         * 
082         * @return XBee protocol description.
083         */
084        public String getDescription() {
085                return description;
086        }
087        
088        /**
089         * Returns the {@code XBeeProtocol} associated to the given ID.
090         * 
091         * @param id The ID of the {@code XBeeProtocol} to retrieve.
092         * 
093         * @return The {@code XBeeProtocol} associated to the given ID.
094         */
095        public static XBeeProtocol get(int id) {
096                if (!lookupTable.containsKey(id))
097                        return UNKNOWN;
098                return lookupTable.get(id);
099        }
100        
101        /**
102         * Determines the XBee protocol based on the given Hardware and firmware 
103         * versions.
104         * 
105         * @param hardwareVersion The hardware version of the protocol to 
106         *                        determine.
107         * @param firmwareVersion The firmware version of the protocol to 
108         *                        determine.
109         * 
110         * @return The XBee protocol corresponding to the given hardware and 
111         *         firmware versions.
112         * 
113         * @see HardwareVersion
114         */
115        public static XBeeProtocol determineProtocol(HardwareVersion hardwareVersion, String firmwareVersion) {
116                if (hardwareVersion == null || firmwareVersion == null || hardwareVersion.getValue() < 0x09 
117                                || HardwareVersionEnum.get(hardwareVersion.getValue()) == null)
118                        return UNKNOWN;
119                switch (HardwareVersionEnum.get(hardwareVersion.getValue())) {
120                case XC09_009:
121                case XC09_038:
122                        return XCITE;
123                case XT09_XXX:
124                case XT09B_XXX:
125                        if ((firmwareVersion.length() == 4 && firmwareVersion.startsWith("8"))
126                                        || (firmwareVersion.length() == 5 && firmwareVersion.charAt(1) == '8'))
127                                return XTEND_DM;
128                        return XTEND;
129                case XB24_AXX_XX:
130                case XBP24_AXX_XX:
131                        if ((firmwareVersion.length() == 4 && firmwareVersion.startsWith("8")))
132                                return DIGI_MESH;
133                        return RAW_802_15_4;
134                case XB24_BXIX_XXX:
135                case XBP24_BXIX_XXX:
136                        if ((firmwareVersion.length() == 4 && firmwareVersion.startsWith("1") && firmwareVersion.endsWith("20")) 
137                                        || (firmwareVersion.length() == 4 && firmwareVersion.startsWith("2")))
138                                return ZIGBEE;
139                        else if (firmwareVersion.length() == 4 && firmwareVersion.startsWith("3"))
140                                return SMART_ENERGY;
141                        return ZNET;
142                case XBP09_DXIX_XXX:
143                        if ((firmwareVersion.length() == 4 && firmwareVersion.startsWith("8") 
144                                        || (firmwareVersion.length() == 4 && firmwareVersion.charAt(1) == '8'))
145                                        || (firmwareVersion.length() == 5 && firmwareVersion.charAt(1) == '8'))
146                                return DIGI_MESH;
147                        return DIGI_POINT;
148                case XBP09_XCXX_XXX:
149                        return XC;
150                case XBP08_DXXX_XXX:
151                        return DIGI_POINT;
152                case XBP24B:
153                        if (firmwareVersion.length() == 4 && firmwareVersion.startsWith("3"))
154                                return SMART_ENERGY;
155                        return ZIGBEE;
156                case XB24_WF:
157                case WIFI_ATHEROS:
158                case SMT_WIFI_ATHEROS:
159                        return XBEE_WIFI;
160                case XBP24C:
161                case XB24C:
162                        if (firmwareVersion.length() == 4 && (firmwareVersion.startsWith("5") || firmwareVersion.startsWith("6")))
163                                return SMART_ENERGY;
164                        else if (firmwareVersion.startsWith("2"))
165                                return RAW_802_15_4;
166                        else if (firmwareVersion.startsWith("9"))
167                                return DIGI_MESH;
168                        return ZIGBEE;
169                case XSC_GEN3:
170                case SRD_868_GEN3:
171                        if (firmwareVersion.length() == 4 && firmwareVersion.startsWith("8"))
172                                return DIGI_MESH;
173                        else if (firmwareVersion.length() == 4 && firmwareVersion.startsWith("1"))
174                                return DIGI_POINT;
175                        return XC;
176                case XBEE_CELL_TH:
177                        return UNKNOWN;
178                case XLR_MODULE:
179                        // This is for the old version of the XLR we have (K60), and it is 
180                        // reporting the firmware of the module (8001), this will change in 
181                        // future (after K64 integration) reporting the hardware and firmware
182                        // version of the baseboard (see the case HardwareVersionEnum.XLR_BASEBOARD).
183                        // TODO maybe this should be removed in future, since this case will never be released.
184                        if (firmwareVersion.startsWith("1"))
185                                return XLR;
186                        else
187                                return XLR_MODULE;
188                case XLR_BASEBOARD:
189                        // XLR devices with K64 will report the baseboard hardware version, 
190                        // and also firmware version (the one we have here is 1002, but this value
191                        // is not being reported since is an old K60 version, the module fw version
192                        // is reported instead).
193                        
194                        // TODO [XLR_DM] The next version of the XLR will add DigiMesh support should be added.
195                        // Probably this XLR_DM and XLR will depend on the firmware version.
196                        
197                        if (firmwareVersion.startsWith("1"))
198                                return XLR;
199                        else
200                                return XLR_MODULE;
201                case XB900HP_NZ:
202                        return DIGI_POINT;
203                case XBP24C_TH_DIP:
204                case XB24C_TH_DIP:
205                case XBP24C_S2C_SMT:
206                        if (firmwareVersion.length() == 4 && (firmwareVersion.startsWith("5") || firmwareVersion.startsWith("6")))
207                                return SMART_ENERGY;
208                        else if (firmwareVersion.startsWith("2"))
209                                return RAW_802_15_4;
210                        else if (firmwareVersion.startsWith("9"))
211                                return DIGI_MESH;
212                        return ZIGBEE;
213                case SX_PRO:
214                case SX:
215                case XTR:
216                        if (firmwareVersion.startsWith("2"))
217                                return XTEND;
218                        else if (firmwareVersion.startsWith("8"))
219                                return XTEND_DM;
220                        else
221                                return SX;
222                case S2D_SMT_PRO:
223                case S2D_SMT_REG:
224                case S2D_TH_PRO:
225                case S2D_TH_REG:
226                        return ZIGBEE;
227                case CELLULAR:
228                        return CELLULAR;
229                default:
230                        return ZIGBEE;
231                }
232        }
233        
234        /*
235         * (non-Javadoc)
236         * @see java.lang.Enum#toString()
237         */
238        @Override
239        public String toString() {
240                return description;
241        }
242}
